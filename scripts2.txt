const hre = require("hardhat");

async function main() {
  const AthletePhotoNFT = await hre.ethers.getContractFactory("AthletePhotoNFT");
  const contract = await AthletePhotoNFT.deploy();
  await contract.deployed();

  console.log("AthletePhotoNFT deployed to:", contract.address);
}

main().catch((error) => {
  console.error(error);
  process.exitCode = 1;
});
